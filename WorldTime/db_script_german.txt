DROP TABLE City;
DROP TABLE Timezone;
DROP TABLE State;
DROP TABLE Country;
DROP TABLE Continent;
DROP TABLE Saved_City;

CREATE TABLE Continent
(
    Continent_ID INTEGER PRIMARY KEY,
    Name VARCHAR(13) NOT NULL    
);

CREATE TABLE Country
(
    Country_ID INTEGER PRIMARY KEY,
    Continent_ID SMALLINT NOT NULL,
    Name VARCHAR(50) NOT NULL,
    Native_Name VARCHAR(200) NULL,
    Phone_Code VARCHAR(10) NOT NULL    
);

CREATE TABLE State
(
    State_ID INTEGER PRIMARY KEY,
    Country_ID INTEGER NOT NULL,
    Name VARCHAR(50) NOT NULL        
);

CREATE TABLE Timezone
(
    Timezone_ID INTEGER PRIMARY KEY,
    Name VARCHAR(50) NOT NULL,
    Offset VARCHAR(6) NOT NULL,
    DST SMALLINT NOT NULL,
    DST_Start VARCHAR(11) NOT NULL,
    DST_End VARCHAR(11) NOT NULL    
);

CREATE TABLE City
(
    City_ID INTEGER PRIMARY KEY,
    State_ID INTEGER NOT NULL,
    Timezone_ID INTEGER NOT NULL,
    Name VARCHAR(100) NOT NULL,
    Native_Name VARCHAR(100) NULL,
    Other_Name VARCHAR(100) NULL,
    Is_Country_Capital SMALLINT NOT NULL,
    Phone_Code VARCHAR(50) NULL,
    Weather_Code VARCHAR(50) NULL    
);

CREATE TABLE Saved_City
(
    City_ID INTEGER PRIMARY KEY
);
